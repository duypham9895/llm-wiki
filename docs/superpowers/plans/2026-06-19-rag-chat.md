# RAG Chat Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A Python Chainlit web chat that answers questions over the A+B enriched PRD vault, with deterministic citations to source PRDs, backed by an incremental ChromaDB index.

**Architecture:** A `chat/` Python sub-project alongside A/B's TypeScript. An offline indexer (`python -m chat.index`) reads the vault, chunks each PRD (body chunks + B's summary), embeds via OpenAI `text-embedding-3-small`, and upserts into a persistent Chroma store with provenance metadata — re-embedding only docs whose `body_hash` changed. A Chainlit app (`chainlit run chat/app.py`) embeds the user's question, queries Chroma top-k, builds a grounded prompt, calls the MiniMax chat model, and streams the answer with a code-built citation block. The LLM boundary (`chat/llm.py`) isolates two credentials: OpenAI for embeddings, MiniMax for chat.

**Tech Stack:** Python 3.10, Poetry, pytest. `chromadb` (vector store), `chainlit` (web chat), `httpx` (HTTP to OpenAI + MiniMax), `pyyaml` (frontmatter). macOS `security` CLI for both API keys.

## Global Constraints

- **Stack:** Python 3.10, Poetry. All C code under `chat/`; tests under `chat/tests/`. This is a SEPARATE suite from A/B's vitest — do NOT touch `package.json`/`tsconfig`/`vitest`. Run tests with `poetry run pytest`.
- **C is read-only over the vault.** It reads `.md` files; it NEVER writes to the vault or to Notion.
- **Two credentials, both from macOS keychain, never logged:**
  - Embeddings: OpenAI key — `security find-generic-password -s ringkas-prd-embed -a openai-api-key -w`. Calls `https://api.openai.com/v1/embeddings`, model `text-embedding-3-small` (1536-dim, probe-verified).
  - Chat answers: MiniMax key — `security find-generic-password -s ringkas-prd-enrich -a llm-api-key -w`. Calls `https://9router-1.dat-nguyen.me/v1/chat/completions`, model `minimax/MiniMax-M3`, with `stream: false` (the router defaults to SSE otherwise — confirmed in sub-project B).
- **Citations are code-built, never LLM-generated.** The Sources block is assembled from retrieved chunks' metadata (title, `source_url`, stem). The LLM writes prose only.
- **Anti-hallucination:** the system prompt instructs the model to answer ONLY from provided context and to say it doesn't know when context is insufficient.
- **Incremental indexing:** Chroma stores each entry's `body_hash`; re-embed a doc only if its vault `body_hash` differs from the stored one. A doc gone from the vault → its entries removed. Re-embedding replaces all of a doc's entries (delete by id, re-add).
- **One bad doc never aborts indexing;** never wipe good Chroma entries on a transient embed failure.
- **Vault layout:** PRD files are `<vault>/PRDs/*.md`, excluding names starting with `_`.

```python
import subprocess


def read_secret(service: str, account: str) -> str:
    """Read a secret from the macOS keychain. Never logs the value."""
    out = subprocess.run(
        ["security", "find-generic-password", "-s", service, "-a", account, "-w"],
        capture_output=True, text=True, check=True,
    )
    return out.stdout.strip()
```

```python
import os
import glob
from dataclasses import dataclass, field
import yaml


@dataclass
class Doc:
    stem: str
    id: str
    title: str
    source_url: str
    status: str
    platform: list = field(default_factory=list)
    tags: list = field(default_factory=list)
    summary: str | None = None
    body_hash: str | None = None
    body: str = ""


def _split_frontmatter(content: str) -> tuple[dict, str]:
    if not content.startswith("---\n"):
        return {}, content
    end = content.find("\n---", 4)
    if end == -1:
        return {}, content
    fm_text = content[4:end + 1]
    body = content[end + 4:]
    if body.startswith("\n"):
        body = body[1:]
    if body.startswith("\n"):
        body = body[1:]
    data = yaml.safe_load(fm_text) or {}
    return data, body


def read_doc(path: str) -> Doc:
    with open(path, "r", encoding="utf-8") as f:
        content = f.read()
    data, body = _split_frontmatter(content)
    sync = data.get("sync", {}) or {}
    llm = data.get("llm", {}) or {}
    stem = os.path.splitext(os.path.basename(path))[0]
    return Doc(
        stem=stem,
        id=str(sync.get("id", stem)),
        title=str(sync.get("title", stem)),
        source_url=str(sync.get("source_url", "")),
        status=str(sync.get("status", "")),
        platform=list(sync.get("platform") or []),
        tags=list(llm.get("tags") or []),
        summary=(llm.get("summary") if llm.get("summary") not in ("", None) else None),
        body_hash=(llm.get("body_hash") if llm.get("body_hash") not in ("", None) else None),
        body=body,
    )


def list_docs(prds_dir: str) -> list[str]:
    paths = glob.glob(os.path.join(prds_dir, "*.md"))
    return sorted(p for p in paths if not os.path.basename(p).startswith("_"))
```

```python
from dataclasses import dataclass, field
from chat.vault import Doc


@dataclass
class Chunk:
    doc_stem: str
    doc_id: str
    title: str
    source_url: str
    status: str
    platform: list = field(default_factory=list)
    tags: list = field(default_factory=list)
    chunk_type: str = "body"
    index: int = 0
    text: str = ""


def _split(text: str, size: int, overlap: int) -> list[str]:
    if not text:
        return []
    if len(text) <= size:
        return [text]
    out = []
    step = max(1, size - overlap)
    i = 0
    while i < len(text):
        out.append(text[i:i + size])
        if i + size >= len(text):
            break
        i += step
    return out


def chunk_doc(doc: Doc, size: int, overlap: int) -> list[Chunk]:
    def base(chunk_type, index, text):
        return Chunk(
            doc_stem=doc.stem, doc_id=doc.id, title=doc.title, source_url=doc.source_url,
            status=doc.status, platform=list(doc.platform), tags=list(doc.tags),
            chunk_type=chunk_type, index=index, text=text,
        )

    chunks: list[Chunk] = []
    if doc.summary:
        chunks.append(base("summary", 0, doc.summary))
    for i, part in enumerate(_split(doc.body, size, overlap)):
        chunks.append(base("body", i, part))
    return chunks
```

```python
import time
import httpx


def _default_post(timeout):
    def post(url, headers, json_body):
        resp = httpx.post(url, headers=headers, json=json_body, timeout=timeout)
        if resp.status_code >= 400:
            err = Exception(f"http {resp.status_code}: {resp.text[:200]}")
            err.status = resp.status_code
            raise err
        return resp.json()
    return post


class LlmClient:
    def __init__(self, cfg, http_post, sleep_fn, max_retries):
        self.cfg = cfg
        self.http_post = http_post
        self.sleep_fn = sleep_fn
        self.max_retries = max_retries

    def _with_retry(self, fn):
        attempt = 0
        while True:
            try:
                return fn()
            except Exception as err:
                status = getattr(err, "status", None)
                retriable = status == 429 or (isinstance(status, int) and status >= 500)
                if not retriable or attempt >= self.max_retries:
                    raise
                self.sleep_fn(min(2 ** attempt * 0.3, 5))
                attempt += 1

    def embed(self, texts):
        def call():
            data = self.http_post(
                f"{self.cfg.openai_base}/embeddings",
                {"content-type": "application/json", "authorization": f"Bearer {self.cfg.openai_key}"},
                {"model": self.cfg.embed_model, "input": texts},
            )
            return [row["embedding"] for row in data["data"]]
        return self._with_retry(call)

    def chat(self, messages):
        def call():
            data = self.http_post(
                f"{self.cfg.minimax_base}/chat/completions",
                {"content-type": "application/json", "authorization": f"Bearer {self.cfg.minimax_key}"},
                {"model": self.cfg.chat_model, "messages": messages, "temperature": 0.2, "stream": False},
            )
            return data["choices"][0]["message"]["content"]
        return self._with_retry(call)


def make_client(cfg, http_post=None, sleep_fn=time.sleep):
    timeout = getattr(cfg, "request_timeout", 60)
    retries = getattr(cfg, "max_retries", 3)
    return LlmClient(cfg, http_post or _default_post(timeout), sleep_fn, retries)
```

```python
import chromadb

COLLECTION = "prds"


class Store:
    def __init__(self, collection):
        self.collection = collection

    @classmethod
    def open(cls, path: str) -> "Store":
        client = chromadb.PersistentClient(path=path)
        collection = client.get_or_create_collection(COLLECTION)
        return cls(collection)

    def upsert(self, chunks, embeddings, body_hash: str) -> None:
        ids, metadatas, documents = [], [], []
        for ch, emb in zip(chunks, embeddings):
            ids.append(f"{ch.doc_stem}:{ch.chunk_type}:{ch.index}")
            metadatas.append({
                "doc_stem": ch.doc_stem, "doc_id": ch.doc_id, "title": ch.title,
                "source_url": ch.source_url, "status": ch.status,
                "platform": ",".join(ch.platform), "tags": ",".join(ch.tags),
                "chunk_type": ch.chunk_type, "body_hash": body_hash,
            })
            documents.append(ch.text)
        self.collection.upsert(ids=ids, embeddings=list(embeddings), metadatas=metadatas, documents=documents)

    def delete_by_doc(self, doc_stem: str) -> None:
        self.collection.delete(where={"doc_stem": doc_stem})

    def stored_hashes(self) -> dict:
        got = self.collection.get(include=["metadatas"])
        out = {}
        for md in got.get("metadatas", []) or []:
            out[md["doc_stem"]] = md["body_hash"]
        return out

    def query(self, embedding, k: int) -> list:
        res = self.collection.query(query_embeddings=[list(embedding)], n_results=k,
                                    include=["documents", "metadatas", "distances"])
        out = []
        docs = (res.get("documents") or [[]])[0]
        metas = (res.get("metadatas") or [[]])[0]
        dists = (res.get("distances") or [[]])[0]
        for text, md, dist in zip(docs, metas, dists):
            out.append({"text": text, "metadata": md, "distance": dist})
        return out
```

```python
import sys
from chat.config import load_config
from chat.keychain import read_secret
from chat.vault import read_doc, list_docs
from chat.chunk import chunk_doc
from chat.store import Store
from chat.llm import make_client


def run_index(cfg, store, embed_fn, read_doc_fn=read_doc, list_docs_fn=list_docs) -> dict:
    stored = store.stored_hashes()
    indexed = skipped = removed = errors = 0
    seen = set()

    for path in list_docs_fn(cfg.prds_dir):
        try:
            doc = read_doc_fn(path)
            seen.add(doc.stem)
            if stored.get(doc.stem) == doc.body_hash and doc.body_hash is not None:
                skipped += 1
                continue
            chunks = chunk_doc(doc, cfg.chunk_size, cfg.chunk_overlap)
            if not chunks:
                skipped += 1
                continue
            embeddings = embed_fn([c.text for c in chunks])
            store.delete_by_doc(doc.stem)
            store.upsert(chunks, embeddings, body_hash=doc.body_hash or "")
            indexed += 1
        except Exception as err:  # one bad doc never aborts the run
            errors += 1
            print(f"error indexing {path}: {err}", file=sys.stderr)

    for stem in list(stored.keys()):
        if stem not in seen:
            store.delete_by_doc(stem)
            removed += 1

    return {"indexed": indexed, "skipped": skipped, "removed": removed, "errors": errors}


def main() -> int:
    import os
    cfg = load_config(os.environ, read_secret)
    store = Store.open(cfg.chroma_path)
    client = make_client(cfg)
    res = run_index(cfg, store, client.embed)
    print(f"indexed {res['indexed']} · skipped {res['skipped']} · removed {res['removed']} · errors {res['errors']}")
    return 1 if res["errors"] else 0


if __name__ == "__main__":
    sys.exit(main())
```

```python
from dataclasses import dataclass


@dataclass
class Retrieved:
    doc_stem: str
    doc_id: str
    title: str
    source_url: str
    text: str
    chunk_type: str


def retrieve(question: str, store, embed_fn, k: int) -> list:
    vec = embed_fn([question])[0]
    rows = store.query(vec, k)
    seen = set()
    out = []
    for r in rows:
        md = r["metadata"]
        stem = md["doc_stem"]
        if stem in seen:
            continue
        seen.add(stem)
        out.append(Retrieved(
            doc_stem=stem, doc_id=md["doc_id"], title=md["title"],
            source_url=md["source_url"], text=r["text"], chunk_type=md["chunk_type"],
        ))
    return out
```

```python
SYSTEM = (
    "You answer questions about Ringkas PRDs using ONLY the provided context. "
    "Cite the PRDs you used by their ID. If the context does not answer the question, "
    "say you don't have a PRD covering that — do not invent. Be concise and direct."

def build_messages(question: str, retrieved: list) -> list:
    context_lines = []
    for r in retrieved:
        context_lines.append(f"[{r.doc_id} · {r.title}] {r.text}")
    context = "\n\n".join(context_lines)
    user = f"Question: {question}\n\nContext:\n{context}"
    return [{"role": "system", "content": SYSTEM}, {"role": "user", "content": user}]


def format_citations(retrieved: list) -> str:
    lines = ["Sources:"]
    seen = set()
    for r in retrieved:
        if r.doc_stem in seen:
            continue
        seen.add(r.doc_stem)
        lines.append(f"- {r.title} — [Notion]({r.source_url}) · [[{r.doc_stem}]]")
    return "\n".join(lines)


def answer(question: str, retrieved: list, chat_fn) -> str:
    if not retrieved:
        return "I don't have a PRD covering that."
    prose = chat_fn(build_messages(question, retrieved))
    return f"{prose}\n\n{format_citations(retrieved)}"
```
